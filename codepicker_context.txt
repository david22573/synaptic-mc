// ==========================================

// Tick represents the incoming game state from Mineflayer
type Tick struct {
	State json.RawMessage
}

// Action represents a single executable step
type Action struct {
	Action    string `json:"action"`
	Target    string `json:"target"`
	Rationale string `json:"rationale"`
}

// LLMPlan represents the multi-step queue generated by the AI
type LLMPlan struct {
	Objective string   `json:"objective"`
	Tasks     []Action `json:"tasks"`
}

// Brain defines the interface for the cognitive layer
type Brain interface {
	EvaluatePlan(ctx context.Context, t Tick) (*LLMPlan, error)
}

// ==========================================
// LLM IMPLEMENTATION
// ==========================================

type LLMBrain struct {
	apiURL string
	model  string
	client *http.Client
	memory MemoryBank
}

// NewLLMBrain initializes the connection to the API
func NewLLMBrain(apiURL, model string, mem MemoryBank) *LLMBrain {
	return &LLMBrain{
		apiURL: apiURL,
		model:  model,
		memory: mem,
		client: &http.Client{Timeout: 30 * time.Second},
	}
}

func (b *LLMBrain) EvaluatePlan(ctx context.Context, t Tick) (*LLMPlan, error) {
	// Safety check: Ensure the API key actually loaded from .env
	apiKey := os.Getenv("OPENROUTER_API_KEY")
	if apiKey == "" {
		return nil, fmt.Errorf("OPENROUTER_API_KEY is not set in environment")
	}

	// 1. Retrieve episodic memory from SQLite
	history, err := b.memory.GetRecentContext(ctx, 10)
	if err != nil {
		log.Printf("[!] Failed to fetch memory: %v", err)
		history = "No recent memory available."
	}

	// 2. Build the RAG system prompt
	systemPrompt := fmt.Sprintf(`You are an autonomous Minecraft survival agent. 
You MUST output a valid JSON object. The "tasks" field MUST be an array of objects, NOT strings.

Example of REQUIRED format:
{
  "objective": "Gather resources",
  "tasks": [
    {
      "action": "mine",
      "target": "oak_log",
      "rationale": "Need wood for a crafting table"
    },
    {
      "action": "retreat",
      "target": "none",
      "rationale": "Move away from danger if encountered"
    }
  ]
}

--- RECENT MEMORY ---
%s
---------------------

Analyze the current game state and your recent memory. Generate a sequential list of tasks to execute.
Valid actions: "attack", "retreat", "mine", "idle".
Target: entity name, block name, or none.`, history)

	payload := map[string]interface{}{
		"model": b.model,
		"messages": []map[string]string{
			{"role": "system", "content": systemPrompt},
			{"role": "user", "content": string(t.State)},
		},
		"response_format": map[string]string{"type": "json_object"},
		"temperature":     0.2, // Low temperature for deterministic planning
	}

	jsonPayload, err := json.Marshal(payload)
	if err != nil {
		return nil, fmt.Errorf("failed to marshal payload: %w", err)
	}

	req, err := http.NewRequestWithContext(ctx, http.MethodPost, b.apiURL, bytes.NewBuffer(jsonPayload))
	if err != nil {
		return nil, fmt.Errorf("failed to create request: %w", err)
	}

	// ==========================================
	// CRITICAL: OPENROUTER AUTH HEADERS
	// ==========================================
	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("Authorization", "Bearer "+apiKey)
	req.Header.Set("HTTP-Referer", "http://localhost:8080")
	req.Header.Set("X-Title", "CraftD Bot Controller")

	startTime := time.Now()
	resp, err := b.client.Do(req)
	if err != nil {
		return nil, fmt.Errorf("API request failed: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		body, _ := io.ReadAll(resp.Body)
		return nil, fmt.Errorf("API HTTP %d: %s", resp.StatusCode, string(body))
	}

	var result struct {
		Choices []struct {
			Message struct {
				Content string `json:"content"`
			} `json:"message"`
		} `json:"choices"`
	}

	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return nil, fmt.Errorf("failed to decode response: %w", err)
	}

	if len(result.Choices) == 0 {
		return nil, fmt.Errorf("no choices returned from API")
	}

	content := result.Choices[0].Message.Content
	log.Printf("[+] AI Latency: %v | Memory Injected: %d bytes\n", time.Since(startTime), len(history))

	var plan LLMPlan
	if err := json.Unmarshal([]byte(content), &plan); err != nil {
		return nil, fmt.Errorf("failed to parse structured JSON: %w\nRaw Content: %s", err, content)
	}

	return &plan, nil
}

package main

import (
	"context"
	"encoding/json"
	"log"
	"sync"
	"time"

	"github.com/gorilla/websocket"
)

// Engine manages the WebSocket lifecycle and the stateful task queue
type Engine struct {
	brain     Brain
	conn      *websocket.Conn
	memory    MemoryBank // Injected SQLite memory
	mu        sync.Mutex
	taskQueue []Action
}

// NewEngine initializes the engine with the brain and database connection
func NewEngine(b Brain, c *websocket.Conn, mem MemoryBank) *Engine {
	return &Engine{
		brain:     b,
		conn:      c,
		memory:    mem,
		taskQueue: make([]Action, 0),
	}
}

// Run blocks and handles the event loop for a connected bot
func (e *Engine) Run(ctx context.Context) {
	defer e.conn.Close()
	var processingMutex sync.Mutex

	for {
		var msg struct {
			Type    string          `json:"type"`
			Payload json.RawMessage `json:"payload"`
		}

		if err := e.conn.ReadJSON(&msg); err != nil {
			log.Println("[-] Bot disconnected or read error:", err)
			return
		}

		// 1. Handle Asynchronous Events from the JS Brainstem
		if msg.Type == "event" {
			var eventPayload struct {
				Event  string `json:"event"`
				Cause  string `json:"cause"`
				Action string `json:"action"` // Used for task completion
			}
			json.Unmarshal(msg.Payload, &eventPayload)

			e.mu.Lock()
			if eventPayload.Event == "panic_retreat" {
				log.Printf("[!] Reflex Triggered: %s. Clearing queue.", eventPayload.Cause)
				e.taskQueue = nil // Drop all planned tasks, survival takes priority
				e.memory.LogEvent(ctx, "PANIC", "Fled from immediate threat: "+eventPayload.Cause)
			} else if eventPayload.Event == "task_complete" {
				log.Printf("[+] Task Complete: %s", eventPayload.Action)
				e.memory.LogEvent(ctx, "COMPLETED", eventPayload.Action)
			}
			e.mu.Unlock()
			continue
		}

		// 2. Handle Game State Ticks
		if msg.Type == "state" {
			e.mu.Lock()
			queueLen := len(e.taskQueue)
			e.mu.Unlock()

			// If we have tasks queued, pop the first one and send it. Skip the LLM.
			if queueLen > 0 {
				e.mu.Lock()
				nextTask := e.taskQueue[0]
				e.taskQueue = e.taskQueue[1:]
				e.mu.Unlock()
				e.execute(nextTask)
				continue
			}

			// Queue is empty. Drop tick if we are already generating a plan.
			if !processingMutex.TryLock() {
				continue
			}

			go func(payload json.RawMessage) {
				defer processingMutex.Unlock()
				evalCtx, cancel := context.WithTimeout(ctx, 25*time.Second) // Slightly longer timeout for local inference
				defer cancel()

				// Ask the LLM to generate a new multi-step plan
				plan, err := e.brain.EvaluatePlan(evalCtx, Tick{State: payload})
				if err != nil {
					log.Printf("[-] Brain evaluation error: %v\n", err)
					return
				}

				// Unpack the plan
				if plan != nil && len(plan.Tasks) > 0 {
					log.Printf("[+] New Plan: %s (%d tasks)", plan.Objective, len(plan.Tasks))
					e.memory.LogEvent(evalCtx, "PLANNING", "Formulated new objective: "+plan.Objective)

					e.mu.Lock()
					firstTask := plan.Tasks[0]
					if len(plan.Tasks) > 1 {
						e.taskQueue = plan.Tasks[1:] // Save the rest of the array to memory
					} else {
						e.taskQueue = nil
					}
					e.mu.Unlock()

					// ONLY pass the single extracted Action to the execute function
					e.execute(firstTask)
				}
			}(msg.Payload)
		}
	}
}

// execute sends a single Action over the WebSocket to the TypeScript client
func (e *Engine) execute(action Action) {
	e.mu.Lock()
	defer e.mu.Unlock()

	payload, _ := json.Marshal(action)
	response := map[string]interface{}{
		"type":    "command",
		"payload": json.RawMessage(payload),
	}

	if err := e.conn.WriteJSON(response); err != nil {
		log.Println("[-] Failed to send command to bot:", err)
	}
}

import pkg from "mineflayer-pathfinder";
const { pathfinder, Movements, goals } = pkg;
import mineflayer from "mineflayer";
import WebSocket from "ws";
import { mineflayer as viewer } from "prismarine-viewer";

// ==========================================
// CONFIGURATION & STATE
// ==========================================
const WS_URL = "ws://localhost:8080/ws";

const bot = mineflayer.createBot({
    host: "localhost",
    port: 25565,
    username: "CraftBot",
    version: "1.19",
});

bot.loadPlugin(pathfinder);

let ws: WebSocket;
let currentTask: string | null = null;
let isBusy = false; // Prevents the bot from spamming the server while working
let combatInterval: NodeJS.Timeout | null = null;
let lastReflexTime = 0;

// ==========================================
// WEBSOCKET LOGIC
// ==========================================
function connectControlPlane() {
    ws = new WebSocket(WS_URL);

    ws.on("open", () => console.log("[+] Connected to Go Control Plane."));

    ws.on("message", (data: Buffer) => {
        try {
            const msg = JSON.parse(data.toString());
            if (msg.type === "command") {
                executeDecision(msg.payload);
            }
        } catch (err) {
            console.error("[-] Failed to parse command from Go:", err);
        }
    });

    ws.on("close", () => {
        console.log("[-] Disconnected from Control Plane. Retrying in 5s...");
        isBusy = false;
        setTimeout(connectControlPlane, 5000);
    });

    ws.on("error", (err) => console.error("WS Error:", err));
}

// ==========================================
// THREAT MATRIX & REFLEXES (Brainstem)
// ==========================================
const THREAT_WEIGHTS: Record<string, number> = {
    warden: 1000,
    creeper: 100,
    skeleton: 20,
    zombie: 10,
    spider: 10,
};

function getThreats() {
    return Object.values(bot.entities)
        .filter((e) => e.type === "mob" || e.type === "hostile")
        .map((e) => {
            const distance = bot.entity.position.distanceTo(e.position);
            const baseThreat = THREAT_WEIGHTS[e.name?.toLowerCase() || ""] || 5;
            // Threat increases exponentially as distance decreases
            const threatScore = baseThreat * (10 / Math.max(distance, 1));

            return {
                id: e.id,
                name: e.name || "unknown",
                distance: parseFloat(distance.toFixed(1)),
                threatScore: Math.round(threatScore),
                position: e.position,
            };
        })
        .sort((a, b) => b.threatScore - a.threatScore)
        .slice(0, 5);
}

function triggerReflexRetreat(threats: any[]) {
    bot.chat("Reflex triggered: Evading immediate threat!");
    const primaryThreat = threats[0];

    // Calculate escape vector
    const dx = bot.entity.position.x - primaryThreat.position.x;
    const dz = bot.entity.position.z - primaryThreat.position.z;
    const length = Math.sqrt(dx * dx + dz * dz);

    const safeX = bot.entity.position.x + (dx / length) * 20;
    const safeZ = bot.entity.position.z + (dz / length) * 20;

    bot.pathfinder.setGoal(
        new goals.GoalNear(safeX, bot.entity.position.y, safeZ, 2),
    );

    // Notify Go to clear the LLM task queue
    if (ws && ws.readyState === WebSocket.OPEN) {
        ws.send(
            JSON.stringify({
                type: "event",
                payload: { event: "panic_retreat", cause: primaryThreat.name },
            }),
        );
    }

    currentTask = null;
    isBusy = true; // Lock the AI from planning while we run away
}

// ==========================================
// EXECUTION & COMPLETION (Cortex)
// ==========================================
function completeTask() {
    if (!currentTask) return;

    console.log(`[+] Task Complete: ${currentTask}`);
    if (ws && ws.readyState === WebSocket.OPEN) {
        ws.send(
            JSON.stringify({
                type: "event",
                payload: { event: "task_complete", action: currentTask },
            }),
        );
    }
    currentTask = null;
    isBusy = false; // Unlocks the bot to ask for the next state
}

function executeDecision(decision: any) {
    console.log(
        `\n[>>] Action: ${decision.action.toUpperCase()} | Target: ${decision.target}`,
    );
    if (decision.rationale)
        console.log(`[>>] Rationale: ${decision.rationale}\n`);

    bot.clearControlStates();
    bot.pathfinder.setGoal(null);
    if (combatInterval) clearInterval(combatInterval);

    currentTask = `${decision.action} ${decision.target}`;
    isBusy = true; // Lock the event loop

    switch (decision.action) {
        case "attack":
            bot.chat(`Engaging ${decision.target}!`);
            const targetEntity = bot.nearestEntity(
                (e) => e.name === decision.target,
            );

            if (targetEntity) {
                bot.pathfinder.setGoal(
                    new goals.GoalFollow(targetEntity, 2),
                    true,
                );

                combatInterval = setInterval(() => {
                    const dist = bot.entity.position.distanceTo(
                        targetEntity.position,
                    );
                    if (dist < 3) {
                        bot.lookAt(
                            targetEntity.position.offset(
                                0,
                                targetEntity.height,
                                0,
                            ),
                        );
                        bot.attack(targetEntity);
                    }

                    // If the entity is dead or despawned, task is complete
                    if (!targetEntity.isValid) {
                        clearInterval(combatInterval!);
                        completeTask();
                    }
                }, 500);
            } else {
                console.log(`[-] Target ${decision.target} not found.`);
                setTimeout(completeTask, 1000); // Fail gracefully
            }
            break;

        case "retreat":
            bot.chat("Tactical retreat!");
            const dx = (Math.random() - 0.5) * 15;
            const dz = (Math.random() - 0.5) * 15;
            const safeX = bot.entity.position.x + dx;
            const safeZ = bot.entity.position.z + dz;
            bot.pathfinder.setGoal(
                new goals.GoalNear(safeX, bot.entity.position.y, safeZ, 2),
            );
            break;

        case "idle":
            setTimeout(completeTask, 2000);
            break;

        case "mine":
            // Note: Placeholder block.
            console.log(`Simulating mining ${decision.target}...`);
            setTimeout(completeTask, 1000);
            break;

        default:
            setTimeout(completeTask, 1000); // Fallback to prevent permanent locks
            break;
    }
}

// Listen for pathfinder completing its goal
bot.on("goal_reached", () => {
    // If we were running away or walking to a block, mark it done
    if (
        currentTask &&
        (currentTask.startsWith("retreat") || currentTask.startsWith("mine"))
    ) {
        completeTask();
    }
});

// ==========================================
// EVENT LOOP
// ==========================================
bot.once("spawn", () => {
    console.log("[+] Bot spawned in the world.");

    const defaultMove = new Movements(bot);
    bot.pathfinder.setMovements(defaultMove);

    connectControlPlane();

    try {
        viewer(bot, {
            port: 3000,
            firstPerson: true,
            viewDistance: 2, // CRITICAL: Restricts rendering to prevent CPU starvation
        });
        console.log("[+] Spectator stream live at http://localhost:3000");
    } catch (err) {
        console.error("[-] Failed to start prismarine-viewer:", err);
    }
});

// 1. High-frequency reflex check (Zero Latency)
bot.on("physicTick", () => {
    const now = Date.now();
    const threats = getThreats();

    // Bypass all logic if we are about to die
    if (
        (threats.length > 0 && threats[0]!.threatScore > 50) ||
        bot.health < 6
    ) {
        if (now - lastReflexTime > 2000) {
            triggerReflexRetreat(threats);
            lastReflexTime = now;
        }
    }
});

// 2. Low-frequency cognitive sync (Networked)
setInterval(() => {
    if (!ws || ws.readyState !== WebSocket.OPEN) return;
    if (isBusy) return; // Wait for the current task to finish

    // Build a rich state object so the LLM has good context
    const state = {
        health: bot.health,
        food: bot.food,
        position: bot.entity.position,
        threats: getThreats(),
        inventory: bot.inventory
            .items()
            .map((i) => ({ name: i.name, count: i.count })),
    };

    ws.send(JSON.stringify({ type: "state", payload: state }));
    isBusy = true; // Lock until the Go server responds
}, 2000);

package main

import (
	"context"
	"log"
	"net/http"
	"os"

	"github.com/gorilla/websocket"
	"github.com/joho/godotenv"
)

const (
	Port         = ":8080"
	DefaultAPI   = "https://openrouter.ai/api/v1/chat/completions"
	DefaultModel = "mistralai/mistral-small-2603"
	DatabasePath = "./bot_memory.db"
)

var upgrader = websocket.Upgrader{
	CheckOrigin: func(r *http.Request) bool { return true },
}

func main() {
	// 1. Load environment variables
	if err := godotenv.Load(); err != nil {
		log.Println("[!] No .env file found. Relying on system environment or defaults.")
	}

	apiURL := os.Getenv("LLM_API_URL")
	if apiURL == "" {
		apiURL = DefaultAPI
	}

	modelName := os.Getenv("LLM_MODEL")
	if modelName == "" {
		modelName = DefaultModel
	}

	// 2. Initialize the modernc/sqlite database
	memory, err := NewSQLiteMemory(DatabasePath)
	if err != nil {
		log.Fatalf("[-] Fatal: could not initialize database: %v", err)
	}
	defer memory.Close()
	log.Printf("[+] Database initialized at %s", DatabasePath)

	// 3. Instantiate the context-aware Brain
	brain := NewLLMBrain(apiURL, modelName, memory)
	log.Printf("[+] Brain wired to %s using model: %s", apiURL, modelName)

	// 4. Set up the WebSocket route
	http.HandleFunc("/ws", func(w http.ResponseWriter, r *http.Request) {
		conn, err := upgrader.Upgrade(w, r, nil)
		if err != nil {
			log.Println("[-] WebSocket upgrade failed:", err)
			return
		}
		log.Printf("[+] Bot connected from %s", r.RemoteAddr)

		engine := NewEngine(brain, conn, memory)

		// Run blocks the handler until the WebSocket connection is closed
		engine.Run(context.Background())

		log.Printf("[-] Bot disconnected: %s", r.RemoteAddr)
	})

	// 5. Start the server
	log.Printf("[=] CraftD Control Plane listening on ws://localhost%s/ws", Port)
	if err := http.ListenAndServe(Port, nil); err != nil {
		log.Fatalf("[-] Server crashed: %v", err)
	}
}

package main

import (
	"context"
	"database/sql"
	"fmt"
	"time"

	_ "modernc.org/sqlite"
)

// MemoryBank defines the interface for our context holder.
type MemoryBank interface {
	LogEvent(ctx context.Context, action, details string) error
	GetRecentContext(ctx context.Context, limit int) (string, error)
	Close() error
}

type SQLiteMemory struct {
	db *sql.DB
}

// NewSQLiteMemory initializes the DB using the CGO-free modernc driver.
func NewSQLiteMemory(dbPath string) (*SQLiteMemory, error) {
	// Note the driver name is now "sqlite", not "sqlite3"
	db, err := sql.Open("sqlite", dbPath)
	if err != nil {
		return nil, fmt.Errorf("failed to open sqlite: %w", err)
	}

	// CRITICAL: Prevent "database is locked" errors in concurrent Go apps.
	// SQLite supports concurrent reads, but only one write at a time.
	db.SetMaxOpenConns(1)

	// Enable Write-Ahead Logging (WAL) for better concurrency and performance
	schema := `
	PRAGMA journal_mode=WAL;
	PRAGMA synchronous=NORMAL;

	CREATE TABLE IF NOT EXISTS events (
		id INTEGER PRIMARY KEY AUTOINCREMENT,
		timestamp DATETIME DEFAULT CURRENT_TIMESTAMP,
		action TEXT,
		details TEXT
	);
	
	CREATE TABLE IF NOT EXISTS spatial_memory (
		id INTEGER PRIMARY KEY AUTOINCREMENT,
		name TEXT UNIQUE,
		x REAL,
		y REAL,
		z REAL
	);
	`
	if _, err := db.Exec(schema); err != nil {
		return nil, fmt.Errorf("failed to apply schema and pragmas: %w", err)
	}

	return &SQLiteMemory{db: db}, nil
}

// LogEvent records something the bot just did or experienced.
func (s *SQLiteMemory) LogEvent(ctx context.Context, action, details string) error {
	query := `INSERT INTO events (action, details) VALUES (?, ?)`
	_, err := s.db.ExecContext(ctx, query, action, details)
	return err
}

// GetRecentContext formats the last N events into a readable string for the LLM prompt.
func (s *SQLiteMemory) GetRecentContext(ctx context.Context, limit int) (string, error) {
	query := `SELECT timestamp, action, details FROM events ORDER BY timestamp DESC LIMIT ?`
	rows, err := s.db.QueryContext(ctx, query, limit)
	if err != nil {
		return "", err
	}
	defer rows.Close()

	var contextStr string
	for rows.Next() {
		var timestamp time.Time
		var action, details string
		if err := rows.Scan(&timestamp, &action, &details); err != nil {
			return "", err
		}
		// Format: [10:45:01] MINED: 3 oak_logs
		contextStr = fmt.Sprintf("[%s] %s: %s\n", timestamp.Format("15:04:05"), action, details) + contextStr
	}

	if err := rows.Err(); err != nil {
		return "", err
	}

	return contextStr, nil
}

func (s *SQLiteMemory) Close() error {
	return s.db.Close()
}
